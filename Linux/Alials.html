pacmanar() {
orphans=($(pacman -Qdtq 2>/dev/null))
if [ ${#orphans[@]} -eq 0 ]; then
  echo "✅ Нет осиротевших пакетов."
else
  echo "🗑 Найдены осиротевшие пакеты:"
  printf '%s\n' "${orphans[@]}"
  read -r "ans?❓ Удалить их? [y/N] "
  if [[ "$ans" =~ ^[Yy]$ ]]; then
    sudo pacman -Rns "${orphans[@]}"
  else
    echo "🚫 Отмена."
  fi
fi
}
